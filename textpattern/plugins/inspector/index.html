   SELECT t.*,
          t.ID,
          unix_timestamp(t.Posted) AS uPosted,
          unix_timestamp(t.Expires) AS uExpires,
          unix_timestamp(t.LastMod) AS uLastMod
     FROM textpattern AS `t`
    WHERE 1=1
      AND t.Type != 'trash'
      AND t.Trash = 0
      AND t.ParentID = 1
      AND (t.Status IN (4))
      AND (t.Posted <= now())
      AND (t.Expires > now() OR t.Expires = '0000-00-00 00:00:00')
 ORDER BY t.Posted DESC
    LIMIT 0, 10
